querySpecification :
 selectClause 
 ( intoClause ) ?
 ( fromClause  )?
 ( (whereClause) =>whereClause )?
 ( (groupByClause) => groupByClause 
  ( havingClause  )?
 )? ->^(SELECTCLAUSE selectClause ^(INTOCLAUSE intoClause)? ^(FROMCLAUSE fromClause)? ^(WHERECLAUSE whereClause)?);

 ;
